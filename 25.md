# 异闻录

<!-- 1. 搜索框与结果容器 -->
<div id="page-search-wrapper" style="margin: 20px 0; padding: 15px; border: 2px dashed #007bff; border-radius: 8px; background: #f8f9fa;">
    <input type="text" id="page-search-input" placeholder="输入关键词，匹配内容将置顶显示..." 
        style="width: 100%; padding: 10px; border: 1px solid #ccc; border-radius: 5px; margin-bottom: 10px;">
    
    <div id="search-results-area" style="display: none;">
        <p style="color: #007bff; font-weight: bold; margin-bottom: 5px;">📍 匹配结果置顶：</p>
        <div id="results-container" style="background: white; padding: 10px; border-left: 4px solid #007bff; max-height: 400px; overflow-y: auto; box-shadow: inset 0 0 5px rgba(0,0,0,0.1);">
            <!-- 匹配到的内容会在这里克隆显示 -->
        </div>
        <hr>
    </div>
</div>

<!-- 2. 核心逻辑 -->
<script>
document.getElementById('page-search-input').addEventListener('input', function() {
    const keyword = this.value.trim();
    const resultsArea = document.getElementById('search-results-area');
    const container = document.getElementById('results-container');
    
    // 清空上次搜索结果
    container.innerHTML = '';
    
    if (keyword.length < 1) {
        resultsArea.style.display = 'none';
        return;
    }

    // 定位正文区域 (根据 GitHub Pages 常用主题适配)
    const content = document.querySelector('article') || document.querySelector('.post-content') || document.body;
    // 选取所有需要搜索的元素（排除搜索框本身）
    const elements = content.querySelectorAll('p, li, h1, h2, h3, h4, h5, h6');
    let hasMatch = false;

    elements.forEach(el => {
        // 避免搜索搜索框区域的内容
        if (document.getElementById('page-search-wrapper').contains(el)) return;

        const text = el.innerText;
        if (text.toLowerCase().includes(keyword.toLowerCase())) {
            hasMatch = true;
            
            // 克隆元素
            const clone = el.cloneNode(true);
            
            // 移除克隆元素的 ID 防止冲突
            if (clone.id) clone.removeAttribute('id');
            
            // 执行高亮替换
            const regex = new RegExp(`(${keyword})`, 'gi');
            clone.innerHTML = clone.innerHTML.replace(regex, '<mark style="background: #ffeb3b; padding: 2px 0;">$1</mark>');
            
            // 添加点击跳转功能（可选，点击置顶项跳转到原文位置）
            clone.style.cursor = 'pointer';
            clone.title = '点击跳转到原文位置';
            clone.onclick = () => el.scrollIntoView({ behavior: 'smooth', block: 'center' });
            
            container.appendChild(clone);
        }
    });

    if (hasMatch) {
        resultsArea.style.display = 'block';
    } else {
        container.innerHTML = '<p style="color: #999;">未找到匹配内容</p>';
        resultsArea.style.display = 'block';
    }
});
</script>

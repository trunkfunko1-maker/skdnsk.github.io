<!-- 1. 搜索框 UI -->
<div id="search-container" style="margin: 20px 0;">
    <input type="text" id="page-search-input" placeholder="搜索本页内容..." 
        style="width: 100%; padding: 10px; border: 1px solid #ccc; border-radius: 5px;">
</div>

<!-- 2. 搜索逻辑 -->
<script>
document.getElementById('page-search-input').addEventListener('input', function() {
    const searchText = this.value.toLowerCase();
    // 这里的 'main' 或 'article' 取决于你 GitHub Pages 主题的正文容器标签
    // 如果不确定，可以使用 'body'，但建议指定正文区域以提高效率
    const contentArea = document.querySelector('article') || document.querySelector('.post-content') || document.body;
    
    // 获取正文中的所有段落、标题、列表项
    const elements = contentArea.querySelectorAll('p, li, h1, h2, h3, h4, h5, h6');

    elements.forEach(el => {
        const text = el.innerText.toLowerCase();
        if (text.includes(searchText)) {
            el.style.display = ""; // 显示匹配项
            el.style.backgroundColor = searchText ? "#fff3cd" : ""; // 匹配时高亮背景（可选）
        } else {
            el.style.display = "none"; // 隐藏不匹配项
        }
    });

    // 如果搜索框为空，恢复所有内容显示
    if (searchText === "") {
        elements.forEach(el => {
            el.style.display = "";
            el.style.backgroundColor = "";
        });
    }
});
</script>
